#include <string>
#include <vector>

using namespace std;

vector<int> solution(int brown, int yellow) {
    vector<int> answer;
    int w = (brown/2)-2;
    for(int i=1;i<w;i++){
        if(yellow == i*(w-i)){
            answer.push_back(w-i+2);
            answer.push_back(i+2);
            break;
        }
    }
    return answer;
}
