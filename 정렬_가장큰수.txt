#include <string>
#include <vector>
#include <algorithm>
using namespace std;

bool compare(string a, string b){		// 비교규칙
    return a+b>b+a;
}

string solution(vector<int> numbers) {
    string answer = "";
    int tmp = 0;
    vector<string> s;
    for(int i=0;i<numbers.size();i++){
        s.push_back(to_string(numbers[i]));
        if(numbers[i] != 0){
            tmp = 1;
        }
    }
    sort(s.begin(), s.end(), compare);
    for(int i=0;i<s.size();i++){
        answer += s[i];
    }
    if(tmp == 0){				// 전부 0인경우
        return "0";
    }
    return answer;
}
